This site demos routing a Ractive app with 2 different layouts. The home page and demo page use the basicLayout. The Sidebar 1 and Sidebar 2 pages use the sidebarLayout.

The router doesn't know the layout exists. It simply takes the module/view that was loaded and attaches it to the document. Here's a simplified main.js using the router to load a Ractive view.

define([], function() {
  'use strict';

  // Load the router
  require(['router'], function(router) {

    // Keep track of the currently loaded view so we can run teardown before loading the new view
    var view;

    router
      .registerRoutes({
        home: { path: '/', moduleId: 'pages/home/homePage' },
        notFound: { path: '*', moduleId: 'pages/notFound/notFoundPage' }
      })
      .on('routeload', function onRouteLoad(View) {
        // When a route loads, render the view and attach it to the document
        var render = function() {
          view = new View({ el: 'body' });
        };

        if (view) {
          view.teardown(render);
        } else {
          render();
        }
      })
      .init(); // Set up event handlers and trigger the initial page load
  });
});

Here's an example Ractive view that uses a layout.

define(function(require) {
  'use strict';
  var Ractive = require('ractive'),
      homeTemplate = require('rv!./homeTemplate'),
      Layout = require('layouts/basicLayout/layout');

  // The HomePage
  var HomePage = Ractive.extend({
    template: homeTemplate
  });

  // The object returned by the home page module is a Layout with the HomePage as a component
  return Layout.extend({
    components: { 'content-placeholder': HomePage }
  });
});
