This site demos routing a Nex app with 2 different layouts. The home page and demo page use the basicLayout. The Sidebar 1 and Sidebar 2 pages use the sidebarLayout.

The router doesn't know the layout exists. It simply takes the module/view that was loaded and attaches it to the document. Here's a simplified main.js using the router to load a Nex view.

define([], function() {
  'use strict';

  // Load the router
  require(['router'], function(router) {
    router
      .registerRoutes({
        home: { path: '/', moduleId: 'pages/home/homePage' },
        notFound: { path: '*', moduleId: 'pages/notFound/notFoundPage' }
      })
      .on('routeload', function onRouteLoad(View, routeArguments) {
        // When a route loads, render the view and attach it to the document
        new View(routeArguments).attachTo('body');
      })
      .init(); // Set up event handlers and trigger the initial page load
  });
});

Here's an example Nex view that uses a layout.

define(function(require) {
  'use strict';
  var Nex = require('nex'),
      Handlebars = require('handlebars'),
      homeTemplate = require('text!./homeTemplate.html'),
      Layout = require('layouts/basicLayout/layout');

  return Nex.defineComponent('home-page', {
    template: Handlebars.compile(homeTemplate),
    layout: Layout
  });
});
