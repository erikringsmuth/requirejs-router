This site demos routing a Angular app with 2 different layouts. The home page and demo page use the basicLayout. The Sidebar 1 and Sidebar 2 pages use the sidebarLayout.

Here's a simplified main.js using the router to load a Angular view.

define([], function() {
    'use strict';

    // Load the router
    require(['router'], function(router) {
      router
        .registerRoutes({
          home: { path: '/', moduleId: 'pages/home/homePage' },
          notFound: { path: '*', moduleId: 'pages/notFound/notFoundPage' }
        })
        .on('routeload', function onRouteLoad(View) {
          // When a route loads, render the view and attach it to the document
          document.querySelector('body').innerHTML = view.layout;
          document.querySelector('content-placeholder').innerHTML = view.template;
        })
        .init(); // Set up event handlers and trigger the initial page load
    });
  });

Here's an example Angular view that uses a layout.

define(function(require) {
  'use strict';
  var app       = require('app'),
      template  = require('text!./homeTemplate.html'),
      Layout    = require('layouts/basicLayout/layout');

  app.controller('HomeController', ['$scope', function ($scope) {
  }]);

  return {
    layout: Layout.template,
    template: template
  };
});
